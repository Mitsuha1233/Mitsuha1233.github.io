function led00()
{
    if(c[0] == 0)
    {
        led[0] = led[0] | (1 << 4);
        c[0] = 1;
    }
    else
    {
        led[0] = led[0]&(~(1<<4));
        c[0] = 0;
    }
    randLED();
}
